129
130 1) 2 Muon tracks (nMuon == 2)
131 2) Neutral state (Sum of Muon_charge == 0)
132 3) Must be isolated (RelIso03 < 0.15)
133 4) High transverse momentum (Muon_pt > 25 GeV)

145 "abs(Muon_eta[0]) < 2.4 && abs(Muon_eta[1]) < 2.4")
146 .Define("Z0_p", [](const ROOT::RVecF &pt, const ROOT::RVecF &eta, const ROOT::RVecF &phi, const ROOT::RVecF &mass){
147 ROOT::Math::PtEtaPhiMVector p_1(pt[0],eta[0],phi[0], mass[0]), p_2(pt[1],eta[1], phi[1], mass[1]);
148 return (ROOT::Math::PtEtaPhiMVector)(p_1 + p_2);}, {"Muon_pt", "Muon_eta", "Muon_phi", "Muon_mass"})
149 .Define("mass", "Z0_p.M()")
150 .Filter("abs(mass- 91.1817) < 15")
151 .Define("pt", "Z0_p.Pt()")
152 .Define("y", "abs(Z0_p.Rapidity())")
153 .Define("phi_eta", [](const ROOT::RVecF &phi, const ROOT::RVecF &eta){
154
155 float dphi = ROOT::VecOps::DeltaPhi(phi[0], phi[1]);
156
157 float deta = eta[0] - eta[1];
158
159 float sinDteta = 1./std::cosh(deta/2);

161 return std::tan((M_PI - dphi)/2.0) * sinDteta;
162
163 }, {"Muon_phi", "Muon_eta"});
